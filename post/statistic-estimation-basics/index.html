Statistic estimation basics

2019-11-22

Compare Z-distribution with T-distribution of a estimated mean

We have a sample of a population of mean \(\mu\), each of the element of the sample is a random variable with the same probability distribution :

\[X_1, X_2, ..., X_3\]

To estimate the mean of the population from the sample, the sample mean is used :

\[\bar{X} = \frac{1}{n}\sum_{i=0}^{n}X_i\]

The central limit theorem says that the random variable \(\bar{X}\) has a normal distribution of mean :

\[\mu_{\bar{X}} = E\left[\bar{X}\right] = E\left[\frac{1}{n}\sum_{i=0}^{n}X_i\right] = \frac{1}{n} \sum_{i=0}^{n}E\left[X_i\right] = \frac{1}{n}\sum_{i=0}^{n}\mu = \mu \]

and variance :

\[\sigma^{2}_{\bar{X}} = E\left[(\bar{X}-\mu)^2\right] = E\left[(\frac{1}{n}\sum_{i=0}^{n}X_i - \mu)^2\right] = E\left[\frac{1}{n^2}\sum_{i=0}^{n}(X_i - \mu)^2\right] = \frac{1}{n^2}E\left[\sum_{i=0}^{n}(X_i - \mu)^2\right] = \frac{1}{n^2}\sum_{i=0}^{n}E\left[(X_i - \mu)^2\right] = \frac{1}{n^2}\sum_{i=0}^{n}\sigma^2 = \frac{\sigma^2}{n}\]

So the distribution of the random variable :

\[Z = \frac{\bar{X}-\mu}{\frac{\sigma}{\sqrt{n}}}\]

is the standard normal distribution \(N(0,1)\).

Unfortunately, the population variance population is rarely available. If the original population is normal or the sample size is large, then the sample variance

\[S^2 = \frac{1}{n-1}\sum_{i=0}^{n}(X_i - \mu)^2\]

is considered to be a good enough estimator of the population variance. So the random variable

\[T = \frac{\bar{X_i}-\mu}{\frac{S}{\sqrt{n}}}\]

has a t-distribution with \(v=n-1\) degree of freedom. The larger the sample size, the more standard distribution and t-distribution are similar, as shown in the following code and graph.

library(ggplot2)
library(cowplot)
## 
## ********************************************************
## Note: As of version 1.0.0, cowplot does not change the
##   default ggplot2 theme anymore. To recover the previous
##   behavior, execute:
##   theme_set(theme_cowplot())
## ********************************************************
library(matrixStats)

#set sample size (n), number of sample (n_samples), mean (mu) and sd (sigma)
n = 10
n_samples = 500
mu = 2
sigma = 5

#generate n_samples samples with n values from a normal distribution 
samples <- matrix(rnorm(n*n_samples, mu, sigma), nrow=n_samples ,ncol=n)

#calculate sample mean and sd
x_ = rowMeans(samples)
s = rowSds(samples)

#calculate the values of the Z and T random variables
Z = (x_-mu)/(sigma/sqrt(n))
T = (x_-mu)/(s/sqrt(n))

values = data.frame(cbind(Z,T))

#plot the density of our samples and the true Z-distribution
p1 <- ggplot(values, aes(x = Z)) + 
      stat_density(geom ="line", 
                   position="identity", 
                   color="red", 
                   linetype="dashed") +
      stat_function(fun=dnorm,
                    color="red",
                    args=list(mean=0, 
                              sd=1)) +
      ggtitle("Z-distribution") +
      theme(plot.title = element_text(hjust=0.5, face="bold", size="15"))

#plot the density of our samples and the true T-distribution
p2 <- ggplot(values, aes(x = T)) +
      stat_density(geom ="line", 
                   position="identity", 
                   color="green", 
                   linetype="dashed") + 
      stat_function(fun = dt,
                    color = "green",
                    args = list(df=n-1)) +
      ggtitle("T-distribution") +
      theme(plot.title = element_text(hjust=0.5, face="bold", size="15"))

plot_grid(p1, p2, ncol = 2, nrow = 1)

Estimating the variance

mu = 10
sigma = 4
n = 100
alpha = 0.10

sample = rnorm(n, mu, sigma)
x_ = mean(sample)
s = sd(sample)

v = n-1
chi_lower_v = qchisq(alpha/2, v)
chi_upper_v = qchisq(1-(alpha/2), v)

chi_lower_b = (n-1)*(s**2)/(chi_upper_v)
chi_upper_b = (n-1)*(s**2)/(chi_lower_v)



p <- plot_var_conf_from_normal(mu, sigma, chi_upper_b, chi_lower_b)

p2 <- ggplot(data.frame(x = c(0,chi_upper_b+30)), aes(x = x)) +
        stat_function(fun = dchisq, args = list(df = v)) +
        stat_function(fun = dchisq, 
                      args = list(df = v),
                      xlim = c(chi_upper_v,chi_upper_b+30),
                      geom = "area",
                      alpha=0.2) +
        geom_vline(xintercept = chi_upper_v,col="red") +
        stat_function(fun = dchisq, args = list(df = v)) +
        stat_function(fun = dchisq, 
                      args = list(df = v),
                      xlim = c(0,chi_lower_b),
                      geom = "area",
                      alpha=0.2) +
        geom_vline(xintercept = chi_lower_b,col="red") +
        labs(x = "Chi-Squared",y="p")
